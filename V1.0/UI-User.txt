<script>
const ADDR = "0x987Be7757F077D42F01C7735986c41Af07A1e19F";
const ABI = [
    "function schedules(address) view returns (uint256 totalAmount, uint256 releasedAmount, uint256 start, uint256 cliff, uint256 duration)",
    "function release() external"
];

let provider, signer, contract, timerInterval;

async function connect() {
    if (!window.ethereum) return alert("Please install MetaMask");
    provider = new ethers.BrowserProvider(window.ethereum);
    await provider.send("eth_requestAccounts", []);
    signer = await provider.getSigner();
    const userAddr = await signer.getAddress();
    contract = new ethers.Contract(ADDR, ABI, signer);
    
    document.getElementById('connectionStatus').innerText = "🟢 Connected: " + userAddr.substring(0,6) + "..." + userAddr.substring(38);
    document.getElementById('connectBtn').innerText = "Switch Account";
    
    loadUserStats(userAddr);
}

async function loadUserStats(addr) {
    try {
        const s = await contract.schedules(addr);
        const total = parseFloat(ethers.formatUnits(s.totalAmount, 18));
        const released = parseFloat(ethers.formatUnits(s.releasedAmount, 18));
        const start = Number(s.start);
        const cliff = Number(s.cliff);
        const duration = Number(s.duration);
        const expiry = start + duration;

        if (total === 0) {
            document.getElementById('statusMsg').innerText = "No vesting schedule found.";
            return;
        }

        const updateClocks = () => {
            const now = Math.floor(Date.now() / 1000);
            
            // Cliff UI logic
            if (now < cliff) {
                document.getElementById('cliffContainer').style.display = 'block';
                document.getElementById('expiryContainer').classList.replace('col-12', 'col-6');
                document.getElementById('cliffCountdown').innerText = formatTime(cliff - now);
            } else {
                document.getElementById('cliffContainer').style.display = 'none';
                document.getElementById('expiryContainer').classList.replace('col-6', 'col-12');
            }

            // Expiry UI logic
            if (now < expiry) {
                document.getElementById('expiryContainer').style.display = 'block';
                document.getElementById('expiryCountdown').innerText = formatTime(expiry - now);
            } else {
                document.getElementById('expiryCountdown').innerText = "FULLY VESTED";
            }

            // Calculations
            let vested = 0;
            if (now >= cliff) {
                vested = (total * (now - start)) / duration;
                if (now >= expiry) vested = total;
            }
            
            const claimable = Math.max(0, vested - released);
            document.getElementById('claimable').innerText = claimable.toFixed(4);
            document.getElementById('total').innerText = total.toFixed(2);
            document.getElementById('pBar').style.width = ((vested / total) * 100) + "%";
            document.getElementById('claimBtn').disabled = claimable <= 0.0001;
        };

        if (timerInterval) clearInterval(timerInterval);
        updateClocks();
        timerInterval = setInterval(updateClocks, 1000);

    } catch (e) {
        console.error(e);
    }
}

function formatTime(diff) {
    const d = Math.floor(diff / 86400);
    const h = Math.floor((diff % 86400) / 3600);
    const m = Math.floor((diff % 3600) / 60);
    const s = Math.floor(diff % 60);
    return d > 0 ? `${d}d ${h}h` : `${h}h ${m}m ${s}s`;
}

document.getElementById('connectBtn').addEventListener('click', connect);
document.getElementById('claimBtn').addEventListener('click', async () => {
    try {
        const tx = await contract.release();
        document.getElementById('statusMsg').innerText = "Claiming tokens...";
        await tx.wait();
        location.reload();
    } catch (e) {
        document.getElementById('statusMsg').innerText = "Claim failed.";
    }
});
</script>
